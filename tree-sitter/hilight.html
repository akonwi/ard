23// Function with Result return type
24fn divide(a: Int, b: Int) Result<Int, Str> {
25 match b == 0 {
26 true => Result::err("Cannot divide by zero"),
27 false => Result::ok(a/b),
28 }
29}
30
31// Function with pattern matching on Result
32fn safe_divide(a: Int, b: Int) Int {
33 match divide(a, b) {
34 ok(result) => result,
35 err(_) => 0
36 }
37}
